contract BlockchainWalletDapp 
{
        mapping (address => uint) public tokensOwned;
        uint public totalTokens;
        uint public balanceTokens;
        uint public tokenPrice;
        address private developerAddress;
        
​
        constructor(uint tokensAmount,uint pricePerToken) public{
            totalTokens = tokensAmount;
            balanceTokens = tokensAmount;
            tokenPrice = pricePerToken;
            developerAddress = msg.sender;
        }
        
        function buyTokens() public payable returns(uint){
            uint tokensToBuy = msg.value/tokenPrice;
            require(tokensToBuy <= balanceTokens);
            tokensOwned[msg.sender] += tokensToBuy;
            balanceTokens -= tokensToBuy;
            return tokensToBuy;
        }
        
        function giveTokens(address addrto,uint number) public {
            require(tokensOwned[msg.sender] >= number);
            tokensOwned[msg.sender] -= number;
            tokensOwned[addrto] += number;
        }
        
        function getContractEther(address payable account) public {
            require(msg.sender == developerAddress);
            account.transfer(address(this).balance);
        }
        
        function showContractEther() public view returns(uint){
            return address(this).balance;
        }

do
